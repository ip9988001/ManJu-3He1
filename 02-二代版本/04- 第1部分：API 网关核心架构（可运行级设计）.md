# ⚙️ 三、核心抽象层（Base Adapter）

## 🔥 base_adapter.py

```python
from abc import ABC, abstractmethod
import asyncio
import time
from typing import Any, Dict


class BaseAdapter(ABC):
    """
    所有 API 适配器统一基类
    """

    def __init__(self, config: Dict):
        self.config = config
        self.max_retries = config.get("max_retries", 3)
        self.timeout = config.get("timeout", 60)

    @abstractmethod
    async def send_request(self, prompt_text: str, **kwargs) -> Any:
        pass

    @abstractmethod
    async def check_status(self, task_id: str) -> Any:
        pass

    @abstractmethod
    async def fetch_result(self, task_id: str) -> Any:
        pass

    @abstractmethod
    async def handle_error(self, response: Any) -> Exception:
        pass

    async def retry(self, func, *args, **kwargs):
        """
        指数退避重试机制
        """
        last_exception = None

        for attempt in range(self.max_retries):
            try:
                return await func(*args, **kwargs)

            except Exception as e:
                last_exception = e
                wait_time = 2 ** attempt

                print(f"[Retry] 第{attempt+1}次失败，{wait_time}s后重试：{str(e)}")

                await asyncio.sleep(wait_time)

        raise last_exception
```

# 🚦 五、并发控制（Token Bucket）

## limiter.py

```python
import asyncio
import time


class AsyncLimiter:
    """
    简单 Token Bucket 限流器
    """

    def __init__(self, rate: int, per: int = 1):
        self.rate = rate
        self.per = per
        self.allowance = rate
        self.last_check = time.time()

    async def acquire(self):
        while True:
            current = time.time()
            time_passed = current - self.last_check
            self.last_check = current

            self.allowance += time_passed * (self.rate / self.per)

            if self.allowance > self.rate:
                self.allowance = self.rate

            if self.allowance >= 1:
                self.allowance -= 1
                return

            await asyncio.sleep(0.05)
```

from collections import defaultdict
import datetime


class CostTracker:

    def __init__(self):
        self.cost_map = defaultdict(float)

    def add_cost(self, provider: str, cost: float):
        self.cost_map[provider] += cost

    def get_total(self):
        return dict(self.cost_map)

    def log(self):
        print("\n=== COST REPORT ===")
        for k, v in self.cost_map.items():
            print(f"{k}: ${v:.4f}")

# 🌐 七、API Gateway 核心调度器

## gateway.py

```python
import asyncio
from typing import Dict
from base.limiter import AsyncLimiter
from core.cost_tracker import CostTracker


class APIGateway:

    def __init__(self):
        self.adapters: Dict = {}
        self.limiters: Dict = {}
        self.cost_tracker = CostTracker()

    def register_adapter(self, name: str, adapter, rate_limit=10):
        self.adapters[name] = adapter
        self.limiters[name] = AsyncLimiter(rate_limit)

    async def call(self, provider: str, prompt: str, **kwargs):
        """
        统一入口
        """

        if provider not in self.adapters:
            raise Exception(f"未注册 provider: {provider}")

        adapter = self.adapters[provider]
        limiter = self.limiters[provider]

        await limiter.acquire()

        async def _request():
            result = await adapter.send_request(prompt, **kwargs)

            # 成本记录（示例）
            self.cost_tracker.add_cost(provider, kwargs.get("cost", 0.001))

            return result

        return await adapter.retry(_request)
```

# 🎯 八、示例 Adapter（DeepSeek LLM）

## deepseek.py

```python
import aiohttp
from base.base_adapter import BaseAdapter
from base.exceptions import APIException, RateLimitError


class DeepSeekAdapter(BaseAdapter):

    def __init__(self, config):
        super().__init__(config)
        self.api_key = config["api_key"]
        self.base_url = "https://api.deepseek.com/v1/chat/completions"

    async def send_request(self, prompt_text: str, **kwargs):

        payload = {
            "model": self.config.get("model", "deepseek-chat"),
            "messages": [
                {"role": "system", "content": "只输出中文"},
                {"role": "user", "content": prompt_text}
            ]
        }

        headers = {
            "Authorization": f"Bearer {self.api_key}"
        }

        async with aiohttp.ClientSession() as session:
            async with session.post(self.base_url, json=payload, headers=headers) as resp:

                if resp.status == 429:
                    raise RateLimitError("rate limit")

                if resp.status != 200:
                    raise APIException(await resp.text())

                data = await resp.json()
                return data["choices"][0]["message"]["content"]

    async def check_status(self, task_id):
        return None

    async def fetch_result(self, task_id):
        return None

    async def handle_error(self, response):
        return APIException(response)
```
