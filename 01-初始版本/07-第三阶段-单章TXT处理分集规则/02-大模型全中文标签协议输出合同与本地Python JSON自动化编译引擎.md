## 三、 Python 后台：纯文本切割与单集物料流转逻辑

大模型将纯文本数据传回后，本地 Python 脚本启动"无脑字符串解剖算法"。它完全不需要任何高深的算力，仅靠行扫描和字符串截取（`.split()` / `.startswith()`），就能在毫秒级内完成原著 TXT 物理切肉与单集物料分流。

### 1. 物理切肉与单集排产文件夹分流逻辑

```python
# 伪代码执行逻辑大盘
def 纯文本解析与物理切肉引擎(大模型返回的纯文本字符串, 当期正文全文本, 下一章正文全文本):
    # 1. 提取分集总数
    总集数_行 = [行 for 行 in 大模型返回的纯文本字符串.split("\n") if "总集数：" in 行][0]
    总集数 = int(总集数_行.split("：")[1])
    本地全章派工单大字典["本章动态分集总数"] = 总集数
    
    # 2. 提取并遍历每一集的独立标签块
    当前原著指针位置 = 0
    for idx in range(1, 总集数 + 1):
        单集字典 = 初始化单集数据字典(idx)
        
        # 利用物理切刀抠出单集文本块
        开始标签 = f"<开始分集_{idx}>"
        结束标签 = f"<结束分集_{idx}>"
        单集标签内容 = 大模型返回的纯文本字符串.split(开始标签)[1].split(结束标签)[0].strip()
        
        # 3. 按行清洗并提取字段干货
        各行数据 = 单集标签内容.split("\n")
        临界资产字符串 = ""
        
        for 行 in 各行数据:
            if 行.startswith("是否跨章吞噬：是"):
                单集字典["跨章滑动焊接层"]["是否激活跨章吞噬"] = True
            elif 行.startswith("文本终点锚点："):
                锚点句子 = 行.split("文本终点锚点：")[1].strip()
            elif 行.startswith("音频台词钩子："):
                单集字典["配音台词钩子文案"] = 行.split("音频台词钩子：")[1].strip()
            elif 行.startswith("画面视觉钩子："):
                单集字典["画面视觉分镜构图"] = 行.split("画面视觉钩子：")[1].strip()
            elif 行.startswith("当前临界资产："):
                临界资产字符串 = 行.split("当前临界资产：")[1].strip()
        
        # 4. 解析资产探针（用中文逗号切开）
        if 临界资产字符串:
            单集字典["本集关键资产探针"] = [资产.strip() for 资产 in 临界资产字符串.split(",")]

        # 5. 【硬核物理切肉动作】
        # 判断本集是不是吃了下一章的肉
        if 单集字典["跨章滑动焊接层"]["是否激活跨章吞噬"]:
            # A 碎肉：把当前章剩下的所有字拿走
            本章碎肉 = 当期正文全文本[当前原著指针位置:]
            # B 鲜肉：去下一章捞取直到锚点句子结束的所有字
            下一章切口位置 = 下一章正文全文本.find(锚点句子) + len(锚点句子)
            下一章鲜肉 = 下一章正文全文本[0:下一章切口位置]
            
            最终单集纯文本内容 = 本章碎肉 + 下一章鲜肉
            单集字典["跨章滑动焊接层"]["下一章物理截止句子"] = 锚点句子
        else:
            # 正常在当期正文里切肉
            本章切口位置 = 当期正文全文本.find(锚点句子) + len(锚点句子)
            最终单集纯文本内容 = 当期正文全文本[当前原著指针位置:本章切口位置]
            当前原著指针位置 = 本章切口位置 # 指针向后滚动，供下一集使用
            
        # 6. 将单集纯文本物料物理落盘
        创建本地排产文件夹(f"第X章微观排产区/ep_{str(idx).zfill(2)}/")
        with open(单集字典["单集生肉文本路径"], "w", encoding="utf-8") as f:
            f.write(最终单集纯文本内容)
            
        # 7. 填入生产大盘
        本地全章派工单大字典["分集详细生产大盘"].append(单集字典)

```
